@startuml

' 读者
entity READER {
  * reader_id : BIGINT <<PK>>
  --
  name       : VARCHAR(40)
  gender     : CHAR(1)
  birthday   : DATE
  phone      : VARCHAR(20)
  email      : VARCHAR(60)
  reg_date   : DATE
  reader_type: VARCHAR(20)
  status     : VARCHAR(20)
}

' 管理员
entity LIBRARIAN {
  * librarian_id : BIGINT <<PK>>
  --
  name       : VARCHAR(40)
  phone      : VARCHAR(20)
  hire_date  : DATE
  status     : VARCHAR(20)
}

' 出版社
entity PUBLISHER {
  * publisher_id : BIGINT <<PK>>
  --
  name    : VARCHAR(80)
  address : VARCHAR(200)
  phone   : VARCHAR(20)
}

' 分类（递归）
entity CATEGORY {
  * category_id : BIGINT <<PK>>
  --
  name      : VARCHAR(80)
  parent_id : BIGINT <<FK>>
}

' 图书基本信息
entity BOOK {
  * book_id     : BIGINT <<PK>>
  --
  isbn         : VARCHAR(20)
  title        : VARCHAR(200)
  author       : VARCHAR(100)
  price        : NUMERIC(10,2)
  publish_date : DATE
  language     : VARCHAR(20)
  publisher_id : BIGINT <<FK>>
  category_id  : BIGINT <<FK>>
}

' 藏书册
entity BOOK_COPY {
  * copy_id : BIGINT <<PK>>
  --
  book_id  : BIGINT <<FK>>
  location : VARCHAR(60)
  status   : VARCHAR(20)
}

' 借阅记录（事件）
entity BORROW {
  * borrow_id   : BIGINT <<PK>>
  --
  reader_id    : BIGINT <<FK>>
  copy_id      : BIGINT <<FK>>
  librarian_id : BIGINT <<FK>>
  borrow_date  : DATE
  due_date     : DATE
  return_date  : DATE
  fine         : NUMERIC(10,2)
}

' 关系定义（修正基数和连线语法）
' READER 1 ---- 0..* BORROW：一个读者可借阅多本/0本，一条借阅记录只属于一个读者
READER "1" -- "0..*" BORROW : borrows

' BOOK_COPY 1 ---- 0..* BORROW：一个藏书册可被借阅多次/0次，一条借阅记录只对应一个藏书册
BOOK_COPY "1" -- "0..*" BORROW : is_borrowed_in

' LIBRARIAN 0..1 ---- 0..* BORROW：一个管理员可处理多条记录，一条记录最多被一个管理员处理
LIBRARIAN "0..1" -- "0..*" BORROW : handles

' BOOK 1 ---- 0..* BOOK_COPY：一本书可有多册/0册，一册书只属于一本书
BOOK "1" -- "0..*" BOOK_COPY : has_copies

' PUBLISHER 1 ---- 0..* BOOK：一个出版社可出版多本书/0本，一本书只属于一个出版社
PUBLISHER "1" -- "0..*" BOOK : publishes

' CATEGORY 1 ---- 0..* BOOK：一个分类可包含多本书/0本，一本书只属于一个分类
CATEGORY "1" -- "0..*" BOOK : classifies

' CATEGORY 1 ---- 0..* CATEGORY（父子分类）：一个父分类可包含多个子分类/0个，一个子分类只属于一个父分类
CATEGORY "1" -- "0..*" CATEGORY : parent_of

@enduml
