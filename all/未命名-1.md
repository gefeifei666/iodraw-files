```mermaid
@startuml
hide circle
hide methods
skinparam classAttributeIconSize 0

' 读者
entity READER {
  * reader_id : BIGINT <<PK>>
  --
  name       : VARCHAR(40)
  gender     : CHAR(1)
  birthday   : DATE
  phone      : VARCHAR(20)
  email      : VARCHAR(60)
  reg_date   : DATE
  reader_type: VARCHAR(20)
  status     : VARCHAR(20)
}

' 管理员
entity LIBRARIAN {
  * librarian_id : BIGINT <<PK>>
  --
  name       : VARCHAR(40)
  phone      : VARCHAR(20)
  hire_date  : DATE
  status     : VARCHAR(20)
}

' 出版社
entity PUBLISHER {
  * publisher_id : BIGINT <<PK>>
  --
  name    : VARCHAR(80)
  address : VARCHAR(200)
  phone   : VARCHAR(20)
}

' 分类（递归）
entity CATEGORY {
  * category_id : BIGINT <<PK>>
  --
  name      : VARCHAR(80)
  parent_id : BIGINT <<FK self>>
}

' 图书基本信息
entity BOOK {
  * book_id     : BIGINT <<PK>>
  --
  isbn         : VARCHAR(20)
  title        : VARCHAR(200)
  author       : VARCHAR(100)
  price        : NUMERIC(10,2)
  publish_date : DATE
  language     : VARCHAR(20)
  publisher_id : BIGINT <<FK>>
  category_id  : BIGINT <<FK>>
}

' 藏书册
entity BOOK_COPY {
  * copy_id : BIGINT <<PK>>
  --
  book_id  : BIGINT <<FK>>
  location : VARCHAR(60)
  status   : VARCHAR(20)
}

' 借阅记录（事件）
entity BORROW {
  * borrow_id   : BIGINT <<PK>>
  --
  reader_id    : BIGINT <<FK>>
  copy_id      : BIGINT <<FK>>
  librarian_id : BIGINT <<FK>>
  borrow_date  : DATE
  due_date     : DATE
  return_date  : DATE
  fine         : NUMERIC(10,2)
}

' 关系（基数在注释中说明）

' READER 1 ---- 0..* BORROW
READER ||--o{ BORROW : borrows

' BOOK_COPY 1 ---- 0..* BORROW
BOOK_COPY ||--o{ BORROW : is_borrowed_in

' LIBRARIAN 0..1 ---- 0..* BORROW
LIBRARIAN ||--o{ BORROW : handles

' BOOK 1 ---- 0..* BOOK_COPY
BOOK ||--o{ BOOK_COPY : has_copies

' PUBLISHER 1 ---- 0..* BOOK
PUBLISHER ||--o{ BOOK : publishes

' CATEGORY 1 ---- 0..* BOOK
CATEGORY ||--o{ BOOK : classifies

' CATEGORY 1 ---- 0..* CATEGORY（父子分类）
CATEGORY ||--o{ CATEGORY : parent_of

@enduml
```
